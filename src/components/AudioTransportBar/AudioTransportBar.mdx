<Playground>
  <Component initialState={{ volume: 70 }}>
  {({ setState, state }) => (
    <AudioTransportBar 
      isPlaying={false}
      volume={state.volume}
      onVolumeChanged={(ev) => {
        setState({volume: parseInt(ev.target.value, 10)});
      }}
    />
  )}
  </Component>
