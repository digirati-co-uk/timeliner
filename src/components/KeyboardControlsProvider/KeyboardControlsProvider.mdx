# Keyboard Controls Provider
Handles the keyboard events

<PropsTable of={KeyboardControlsProvider} />

<Playground>
  <Component initialState={{ lastKey: '' }}>
  {({ setState, state }) => {
    return (
      <div>
        <KeyboardControlsProvider
          commands={{
            'default': (keyCombo, key, ev) => setState({lastKey: keyCombo}),
            'alt+a': () => setState({lastKey: 'special alt a handler'}),
          }}
        />
        <p>Last key pressed:<b>{state.lastKey}</b></p>
      </div>
    );
  }}
  </Component>
